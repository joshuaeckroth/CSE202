Newton's method for finding roots

from cmath import *

def newton(x):
    return -2 * sin(cos(sqrt(x)))

def dnewton(x):
    return cos(cos(sqrt(x)))*sin(sqrt(x)) / sqrt(x)

eps = 0.0000001
xold = 2
xnew = xold - newton(xold)/dnewton(xold)
while abs(xold - xnew) > eps:
    xold = xnew
    xnew = xold - newton(xold)/dnewton(xold)

print xnew

Double sum

m = 100
n = 100
sum = 0
for i in range(1, m):
    for j in range(1, n):
        sum += (i+j)**2

print sum

Stars

width = 30
output = []
for i in range(1, width + 1, 2):
    for j in range(0, (width - i) / 2):
        output.append(" ")
    for j in range(0, i):
        output.append("*")
    if i != (width - 1):
        output.append("\n")

print "".join(output)
