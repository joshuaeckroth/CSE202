Newton's method for finding roots

In the file newton.m:

function [y] = newton(x)
y = -2 * sin(cos(sqrt(x)));

In the file dnewton.m:

function [d] = dnewton(x, eps)
eps2 = eps / 1000;
xs = (x - eps):eps2:(x + eps);
y = newton(xs);
dx = diff(xs);
dy = diff(y);
z = dx./dy;
d = z(500:500);

Now, in Matlab:

eps = 0.0000001;
xold = 2;
xnew = xold - newton(xold)/dnewton(xold, eps);
while abs(xold - xnew) > eps
    xold = xnew;
    xnew = xold - newton(xold)/dnewton(xold, eps);
end

Double sum

m = 100;
n = 100;
sum = 0;
for i = 1:m
    for j = 1:n
        sum = sum + (i+j)^2;
    end
end
disp(sum)

Stars

width = 30;
for i = 1:2:width
    for j = 0:((width - i) / 2 - 1)
        fprintf(' ');
    end
    for j = 0:(i-1)
        fprintf('*');
    end
    if i ~= width
        fprintf('\n');
    end
end
